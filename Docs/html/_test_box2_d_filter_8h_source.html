57 double calculateVariance(const Image &image) {
58 unsigned char *data = image.getData();
59 int totalPixels = image.getWidth() * image.getHeight();
60 int channels = image.getChannels();
61
62 // Calculate the average pixel value
63 double sum = 0.0;
64 for (int i = 0; i < totalPixels; ++i) {
65 // Only consider the first channel for simplicity
66 sum += data[i * channels];
67 }
68 double mean = sum / totalPixels;
69
70 // Calculate the variance
71 double variance = 0.0;
72 for (int i = 0; i < totalPixels; ++i) {
73 variance += std::pow(data[i * channels] - mean, 2);
74 }
75 variance /= totalPixels;
76
77 return variance;
78 }
