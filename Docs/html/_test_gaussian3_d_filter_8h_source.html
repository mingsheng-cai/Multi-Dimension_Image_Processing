53 std::vector<double> generateGaussianKernel(double sigma, int kernelSize) const {
54 std::vector<double> kernel(kernelSize);
55 double sum = 0.0;
56 int radius = kernelSize / 2;
57 double sigmaSquare2 = 2 * sigma * sigma;
58
59 // Compute kernel values based on Gaussian function
60 for (int i = -radius; i <= radius; ++i) {
61 double value = (std::exp(-(i * i) / sigmaSquare2)) / (std::sqrt(M_PI * sigmaSquare2));
62 kernel[i + radius] = value;
63 sum += value;
64 }
65
66 // Normalize kernel values
67 for (double &value : kernel) {
68 value /= sum;
69 }
70
71 return kernel;
72 }

188 void analyzeVolumeSmoothing(const Volume& volume, const std::vector<unsigned char>& originalData) {
189 const unsigned char* filteredData = volume.getData();
190 int width = volume.getWidth();
191 int height = volume.getHeight();
192 int depth = volume.getDepth();
193 size_t dataSize = width * height * depth;
194
195 // Calculate mean and variance for original and filtered data
196 double meanOriginal = std::accumulate(originalData.begin(), originalData.end(), 0.0) / dataSize;
197 double meanFiltered = std::accumulate(filteredData, filteredData + dataSize, 0.0) / dataSize;
198
199 double varianceOriginal = 0.0, varianceFiltered = 0.0;
200 for (size_t i = 0; i < dataSize; ++i) {
201 varianceOriginal += (originalData[i] - meanOriginal) * (originalData[i] - meanOriginal);
202 varianceFiltered += (filteredData[i] - meanFiltered) * (filteredData[i] - meanFiltered);
203 }
204 varianceOriginal /= dataSize;
205 varianceFiltered /= dataSize;
206
207 // Verify that the filtered data has lower variance than the original data
208 assert(varianceFiltered < varianceOriginal && "Filtered data should have lower variance than original.");
209
210 // Verify that the filtered data is within the valid intensity range
211 for (size_t i = 0; i < dataSize; ++i) {
212 assert(filteredData[i] <= 255 && "Filtered data contains values out of range.");
213 }
214
215 std::cout << "Gaussian3DFilter variance analysis passed: Original variance = " << varianceOriginal
216 << ", Filtered variance = " << varianceFiltered << std::endl;
