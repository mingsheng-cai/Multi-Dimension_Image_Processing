48 double calculateHistogramStdDev(const std::vector<int> &histogram) {
49 int sum = std::accumulate(histogram.begin(), histogram.end(), 0);
50 double mean = static_cast<double>(sum) / histogram.size();
51
52 double variance = 0.0;
53 for (int value: histogram) {
54 variance += (value - mean) * (value - mean);
55 }
56 variance /= histogram.size();
57
58 return std::sqrt(variance);
59 }
