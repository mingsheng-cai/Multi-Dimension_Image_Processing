41 double calculateVariance(const Image &img) {
42 double sum = 0.0;
43 double sumSquared = 0.0;
44 int pixelCount = img.getWidth() * img.getHeight();
45 for (int i = 0; i < pixelCount; ++i) {
46 double pixelValue = static_cast<double>(img.getData()[i]);
47 sum += pixelValue;
48 sumSquared += pixelValue * pixelValue;
49 }
50 double mean = sum / pixelCount;
51 double variance = (sumSquared / pixelCount) - (mean * mean);
52 return variance;
53 }
